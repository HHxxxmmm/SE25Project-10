import React, { useState, useEffect } from 'react';
import generateMockTrainData from './mock/Train'; // 调整实际路径

const TrainSchedulePage = () => {
  const [trains, setTrains] = useState([]);

  useEffect(() => {
    // 同步获取 Mock 数据
    const mockData = generateMockTrainData();
    setTrains(mockData.news);
    
    // 如果希望模拟网络请求延迟：
    // setTimeout(() => {
    //   setTrains(mockData.news);
    // }, 500);
  }, []);

  // 辅助函数：获取席别名称
  const getSeatType = (type) => {
    const types = {
      1: '头等座',
      2: '商务座',
      3: '二等座',
      4: '无座'
    };
    return types[type] || '未知席别';
  };

  return (
    <div className="train-schedule">
      <h1>列车时刻表</h1>
      
      {trains.length === 0 ? (
        <p>加载中...</p>
      ) : (
        <div className="train-list">
          {trains.map((train, index) => (
            <div key={index} className="train-card">
              <div className="train-header">
                <h2>{train.train_id}</h2>
                <span>{train.t_from} → {train.t_to}</span>
              </div>
              
              <div className="train-times">
                <p>发车: {train.t_start_time}</p>
                <p>到达: {train.t_end_time}</p>
              </div>
              
              <div className="train-route">
                <p>途经站点 ({train.t_station_number}站):</p>
                <ul>
                  {train.t_path.map((station, idx) => (
                    <li key={idx}>{station}</li>
                  ))}
                </ul>
              </div>
              
              <div className="train-seats">
                <h3>席别库存</h3>
                <div className="seat-grid">
                  {train.seat.map((seatType, idx) => (
                    <div key={idx} className="seat-item">
                      <span>{getSeatType(seatType)}</span>
                      <strong>{train.seat_number[idx]}张</strong>
                    </div>
                  ))}
                </div>
              </div>
            </div>
          ))}
        </div>
      )}
    </div>
  );
};

export default TrainSchedulePage;
